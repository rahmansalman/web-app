contains the database connectivity information. 
    define('DB_HOST', 'localhost'); // database host

    define('DB_PASS', 'secret'); // database password

Database tables

The application has two tables:

  1. applicants
    : This table holds all the information about applicant data. Each field on the form is saved into this table except for the 
    image_url
    field which holds a relative url to the uploaded image
  2. unique_codes
    : This table holds all the unique codes used to login before applying. These unique codes can only be used once for an application.
            CREATE TABLE `applicants` (
              `id` int(10) UNSIGNED NOT NULL,
              `access_code` varchar(20) DEFAULT NULL,
              `firstname` varchar(50) NOT NULL,
              `lastname` varchar(50) NOT NULL,
              `address` text,
              `marital_status` varchar(20) NOT NULL,
              `edu_background` varchar(40) NOT NULL,
              `best_subjects` varchar(255) NOT NULL,
              `religion` varchar(20) NOT NULL,
              `state_of_origin` varchar(20) NOT NULL,
              `date_of_birth` date DEFAULT NULL,
              `image_url` varchar(255) NOT NULL
            ) ENGINE=InnoDB DEFAULT CHARSET=latin1;
        
            CREATE TABLE `unique_codes` (
              `id` int(11) NOT NULL,
              `access_code` varchar(10) NOT NULL,
              `used` tinyint(4) NOT NULL DEFAULT '0',
              `created` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
            ) ENGINE=InnoDB DEFAULT CHARSET=latin1;
